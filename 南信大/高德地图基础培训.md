# 地图的创建

```html
<body>
  <div id="container"></div>
  <!-- 加载地图JSAPI脚本 -->
  <script src="https://webapi.amap.com/maps?v=1.4.15&key=e4155ab4d61baff837231a88292d1b77"></script>
  <script>
    var map = new AMap.Map('container', {
      resizeEnable: true, //是否监控地图容器尺寸变化
      zoom: 11, //初始化地图层级
      center: [116.397428, 39.90923] //初始化地图中心点
    });
  </script>
</body>
```



# 添加/移除自定义Marker

```html
<body>
  <div id="container"></div>
  <script type="text/javascript"
    src="https://webapi.amap.com/maps?v=1.4.15&key=e4155ab4d61baff837231a88292d1b77"></script>
  <script type="text/javascript">
    var position = new AMap.LngLat(116.397428, 39.90923);

    // 创建地图实例
    var map = new AMap.Map("container", {
      zoom: 5,
      center: position,
      resizeEnable: true
    });

    // 点标记显示内容，HTML要素字符串
    var markerContent = '' +
      '<div class="custom-content-marker">' +
      '   <div class="close-btn" onclick="clearMarker()">X</div>' +
      '</div>';
    var markerContent2 = '' +
      '<div class="custom-content-marker">' +
      '   <div class="close-btn" onclick="clearMarker2()">X</div>' +
      '</div>';

    var marker = new AMap.Marker({
      position: position,
      // 将 html 传给 content
      content: markerContent,
      // 以 icon 的 [center bottom] 为原点
      offset: new AMap.Pixel(-13, -30)
    });

    var marker2 = new AMap.Marker({
      position: position,
      // 将 html 传给 content
      content: markerContent2,
      // 以 icon 的 [center bottom] 为原点
      offset: new AMap.Pixel(-230, -30)
    });

    // 将 markers 添加到地图
    map.add(marker);
    map.add(marker2);

    // 清除 marker
    function clearMarker () {
      map.remove(marker);
    }

    // 清除 marker2
    function clearMarker2 () {
      map.remove(marker2);
    }
  </script>
</body>
```



# 点（标记）、线、面

```html
<body>
  <div id="container"></div>
  <div class="input-card">
    <h4>添加覆盖物</h4>
    <div class="input-item">
      <input id="addOneOverlay" type="button" class="btn" value="添加点（标记）" />
      <input id="addSeveralOverlay" type="button" class="btn" value="添加线、面" />
    </div>
  </div>
  <script type="text/javascript"
    src="https://webapi.amap.com/maps?v=1.4.15&key=e4155ab4d61baff837231a88292d1b77"></script>
  <script type="text/javascript">
    // 创建地图实例
    var map = new AMap.Map("container", {
      zoom: 13,
      center: [116.43, 39.92],
      resizeEnable: true
    });

    // 创建标记点覆盖物
    var marker = new AMap.Marker({
      position: new AMap.LngLat(116.39, 39.92),
      // icon: 'https://img.tukuppt.com/png_preview/00/33/47/nAvN4zrRH2.jpg!/fw/780',
      offset: new AMap.Pixel(-13, -30)
    });

    var polylinePath = [
      new AMap.LngLat(116.41, 39.93),
      new AMap.LngLat(116.43, 39.91)
    ];

    // 创建线覆盖物
    var polyline = new AMap.Polyline({
      path: polylinePath
    });

    var polygonPath = [
      new AMap.LngLat(116.46, 39.93),
      new AMap.LngLat(116.44, 39.91),
      new AMap.LngLat(116.49, 39.91)
    ];

    // 创建面覆盖物
    var polygon = new AMap.Polygon({
      path: polygonPath
    });

    var addOneOverlayButton = document.getElementById('addOneOverlay');
    var addSeveralOverlayButton = document.getElementById('addSeveralOverlay');

    // 添加点（标记）
    addOneOverlayButton.onclick = function () {
      map.add(marker);
    };

    // 添加线、面
    addSeveralOverlayButton.onclick = function () {
      map.add([polyline, polygon]);
    };
  </script>
</body>
```
